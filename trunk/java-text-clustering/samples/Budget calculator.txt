        function ValidateInput(ControlName, Validator) {
            var objValue = document.getElementById(ControlName);
            var objValidator = document.getElementById(Validator);
            var retval = true;
            var charpos = objValue.value.search("[^0-9]");
            if (objValue.value.length> 0 && charpos>= 0) {
                retval = false;
                objValidator.style.visibility = 'visible';
            }
            else {
                objValidator.style.visibility = 'hidden';
            }
            return retval;
        }
